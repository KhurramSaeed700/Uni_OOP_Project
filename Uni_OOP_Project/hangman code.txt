#include<iostream>

void printTitle(string message, bool printTop, bool printBottom) {
    if (printTop) {
        cout << "+--------------------------------+\n";
        cout << "|";
    }
    bool front = true;
    for (int i = message.length(); i < 32; i++) {
        if (front) {
            message = " " + message;
        }
        else
        {
            message = message + " ";
        }
        front = !front;
    }
    cout << message.c_str();

    if (printBottom) {
        cout << "|" << endl;
        cout << "+--------------------------------+\n";
    }

}

void printLetters(string input, char from, char to) {

    string s;
    for (char i = from; i < to; i++) {
        if (input.find(i) == string::npos) // if input string dose't contain letter then
        {
            s += i;
            s += " ";
        }
        else {
            s += " ";

        }
    }
    printTitle(s, true, true);
}
